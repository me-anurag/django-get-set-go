# ---- CREATE A PROJECT FOLDER ----
mkdir myproject
cd myproject

# ---- CREATE VIRTUAL ENVIRONMENT ----
python -m venv venv
# This creates a venv/ folder with an isolated Python inside

# ---- ACTIVATE VIRTUAL ENVIRONMENT (PowerShell) ----
.\venv\Scripts\Activate.ps1

# You will see (venv) at the start of your terminal line:
# (venv) PS C:\Users\you\myproject>

# ---- IF YOU GET AN ERROR: "execution of scripts is disabled" ----
Set-ExecutionPolicy -ExecutionPolicy RemoteSigned -Scope CurrentUser
# Then try activating again

# ---- DEACTIVATE WHEN DONE ----
deactivate

# ---- WHY VIRTUAL ENVIRONMENT? ----
# Without venv → all packages install globally → projects conflict
# With venv    → each project has its own packages, own Django version
# RULE: Always activate venv before doing ANYTHING in your project
