# ---- blog/models.py ----

from django.db import models
from django.contrib.auth.models import User
from django.urls import reverse
from django.utils import timezone


class Category(models.Model):
    name = models.CharField(max_length=100)
    slug = models.SlugField(unique=True)

    def __str__(self):
        return self.name

    class Meta:
        verbose_name_plural = 'Categories'
        ordering = ['name']


class Post(models.Model):

    # ---- STATUS CHOICES ----
    DRAFT     = 'draft'
    PUBLISHED = 'published'
    STATUS_CHOICES = [
        (DRAFT, 'Draft'),
        (PUBLISHED, 'Published'),
    ]

    # ---- FIELDS ----
    title      = models.CharField(max_length=200)
    slug       = models.SlugField(unique=True)
    body       = models.TextField()
    summary    = models.CharField(max_length=300, blank=True)  # optional
    cover      = models.ImageField(upload_to='covers/', blank=True, null=True)
    status     = models.CharField(max_length=10, choices=STATUS_CHOICES, default=DRAFT)
    views      = models.PositiveIntegerField(default=0)
    created_at = models.DateTimeField(auto_now_add=True)  # set once on create
    updated_at = models.DateTimeField(auto_now=True)      # updated on every save
    published  = models.DateTimeField(null=True, blank=True)

    # ---- RELATIONSHIPS ----
    author     = models.ForeignKey(User, on_delete=models.CASCADE, related_name='posts')
    category   = models.ForeignKey(Category, on_delete=models.SET_NULL, null=True)
    tags       = models.ManyToManyField('Tag', blank=True)  # string ref avoids import order issues

    def __str__(self):
        return self.title

    def get_absolute_url(self):
        return reverse('blog:detail', kwargs={'slug': self.slug})

    class Meta:
        ordering = ['-created_at']   # newest first


class Tag(models.Model):
    name = models.CharField(max_length=50)
    slug = models.SlugField(unique=True)

    def __str__(self):
        return self.name


# ---- FIELD REFERENCE ----
# CharField(max_length)         → short text
# TextField()                   → long text
# IntegerField()                → whole number
# FloatField()                  → decimal
# DecimalField(max_digits, decimal_places) → exact decimal (money)
# BooleanField()                → True/False
# DateField()                   → date only
# DateTimeField()               → date + time
# EmailField()                  → validates email format
# URLField()                    → validates URL
# SlugField()                   → URL-friendly text (letters, numbers, hyphens)
# ImageField(upload_to)         → requires Pillow: pip install Pillow
# FileField(upload_to)          → any file
# JSONField()                   → stores JSON (Django 3.1+)
# ForeignKey(Model, on_delete)  → many-to-one
# ManyToManyField(Model)        → many-to-many
# OneToOneField(Model)          → one-to-one

# on_delete options:
# CASCADE     → delete related objects too
# SET_NULL    → set to null (field must have null=True)
# SET_DEFAULT → set to default value
# PROTECT     → prevent deletion if related objects exist
# DO_NOTHING  → do nothing (dangerous)
